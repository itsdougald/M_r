1. Configure network
Go through System > Preferences > Network Connections > System eth0

#vi /etc/sysconfig/network-scripts/ifcfg-eth0

DEVICE=eth0
IPADDR="192.168.1.4"
BOOTPROTO=none
TYPE="Ethernet"
#UUID="21050187-45f2-410f-87f5-1c54e621af94"
ONBOOT="yes"
NM_CONTROLLED="yes"
IPADDR="192.168.1.4"
NETMASK=255.255.255.0
HWADDR=00:0C:29:25:32:7B
PREFIX=24
GATEWAY=192.168.1.254
DNS1=192.168.1.254
DEFROUTE=yes
IPV4_FAILURE_FATAL=yes
IPV6INIT=no
NAME="System eth0"
UUID=5fb06bd0-0bb0-7ffb-45f1-d6edd65f3e03
LAST_CONNECT=1396084050


#vi /etc/sysconfig/network

ETWORKING=yes
HOSTNAME=vm-web-zebedee
